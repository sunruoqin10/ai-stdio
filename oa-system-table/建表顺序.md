-- 19. 年假余额表（依赖sys_employee）

-- 20. 年假使用记录表（依赖sys_employee和approval_leave_request）
CREATE TABLE approval_leave_usage_log (...);

-- 21. 节假日表（无依赖）

-- 22. 费用报销表（依赖sys_employee和sys_department）
CREATE TABLE approval_expense (...);

-- 23. 费用明细表（依赖approval_expense）
CREATE TABLE approval_expense_item (...);

-- 24. 发票表（依赖approval_expense）
CREATE TABLE approval_expense_invoice (...);

-- 25. 打款记录表（依赖approval_expense）
CREATE TABLE approval_expense_payment (...);

-- 26. 会议室表（无依赖）

-- 27. 会议室预定表（依赖admin_meeting_room和sys_employee）
CREATE TABLE admin_meeting_booking (...);
